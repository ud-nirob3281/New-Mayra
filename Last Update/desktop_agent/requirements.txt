# MYRAA Desktop Control Agent — runtime dependencies.
#
# Each tool module imports its optional dependency lazily (try/except), so a
# missing package only disables that one tool — the agent still boots and the
# /health endpoint still reports. The packages below cover the full tool set.

# Core HTTP server
fastapi
uvicorn
pydantic

# System control
psutil            # CPU/RAM/disk/temp metrics, process listing (tools_system, tools_applications)
pyautogui         # mouse/keyboard synthesis (tools_clipboard fallback, tools_windows)
pycaw             # precise Windows master-volume scalar control (tools_pc)
comtypes          # required by pycaw for COM activation (tools_pc)

# Brightness
screen_brightness_control  # tools_pc; degrades to WMI/PowerShell if absent

# Browser automation (Playwright headed Chromium)
playwright        # tools_browser — opens/navigates/clicks/types/scrolls & media control

# Screenshots & OCR (best-effort; degrade gracefully if missing)
mss               # fast multi-monitor screen capture (tools_screenshot, tools_visual)
Pillow            # image handling (tools_screenshot, tools_visual)
numpy             # array ops for OpenCV preprocessing (tools_visual)
pytesseract       # OCR via Tesseract (tools_visual) — needs Tesseract-OCR installed on the OS

# Optional OCR/ML fallbacks (agent degrades gracefully without these)
opencv-python     # adaptive-threshold preprocessing in OCR pipeline (tools_visual)
easyocr           # PyTorch-based OCR fallback when Tesseract fails (tools_visual)
                  # NOTE: easyocr pulls in PyTorch (~200-400MB). Only installed when
                  # the user opts in. The agent works fine without it.

# Fuzzy text matching (optional — difflib fallback exists)
rapidfuzz         # fast Levenshtein fuzzy matching for OCR text correction (tools_visual)
